\begin{example}
 Use a linear approximation of $f(x)=\sin(x)$
 at $x=0$ to approximate $\sin(0.3)$.
  
 \begin{explanation}
  To start, write
  \[
   \frac{d}{dx} f(x) = \answer{\cos(x)},
  \]
  so our linear approximation is
  \begin{align*}
   \l(x) &= \answer{\cos(0)}\cdot(x-0) + 0\\
         &= x.
  \end{align*}
 \end{explanation}
\end{example}
